import { serve } from "https://deno.land/std@0.168.0/http/server.ts"

const corsHeaders = {
    'Access-Control-Allow-Origin': '*',
    'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
}

serve(async (req) => {
    if (req.method === 'OPTIONS') {
        return new Response('ok', { headers: corsHeaders })
    }

    try {
        const FLW_SECRET_KEY = Deno.env.get('FLW_SECRET_KEY')
        if (!FLW_SECRET_KEY) {
            throw new Error('Flutterwave Secret Key is not set')
        }

        // 1. Destructure all fields including redirect_url, eventId, and message!
        const { amount, email, currency, name, user_id, redirect_url, eventId, message } = await req.json()

        if (!amount || !email || !redirect_url) {
            throw new Error('Missing required fields: amount, email, or redirect_url')
        }

        const tx_ref = `tx_${user_id || 'guest'}_${Date.now()}`

        // 2. Call Flutterwave Standard API (v3)
        const flwResponse = await fetch('https://api.flutterwave.com/v3/payments', {
            method: 'POST',
            headers: {
                'Authorization': `Bearer ${FLW_SECRET_KEY}`,
                'Content-Type': 'application/json',
            },
            body: JSON.stringify({
                tx_ref,
                amount,
                currency: currency || 'NGN',
                redirect_url, // Now correctly defined!
                customer: {
                    email,
                    name: name || 'Supabase User',
                },
                meta: {
                    consumer_id: user_id,
                    eventId, // Passed along for verification later
                    message, // Passed along for verification later
                },
                customizations: {
                    title: 'Relive Inc.',
                    logo: 'https://your-logo-url.com/logo.png',
                },
            }),
        })

        const flwData = await flwResponse.json()

        if (flwData.status !== 'success') {
            console.error('Flutterwave Error:', flwData)
            throw new Error(flwData.message || 'Failed to initialize payment')
        }

        return new Response(
            JSON.stringify({
                data: {
                    link: flwData.data.link,
                    tx_ref
                }
            }),
            { headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
        )

    } catch (error) {
        return new Response(
            JSON.stringify({ error: error.message }),
            { headers: { ...corsHeaders, 'Content-Type': 'application/json' }, status: 400 }
        )
    }
})
